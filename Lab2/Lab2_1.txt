    EmpNo INT PRIMARY KEY,
    EmpName VARCHAR(50) NOT NULL,
    Gender CHAR(1) NOT NULL CHECK (Gender IN ('M', 'F')),
    Salary DECIMAL(10,2) NOT NULL,
    Address VARCHAR(100) NOT NULL,
    DNo INT
);

    DeptNo INT PRIMARY KEY,
    DeptName VARCHAR(50) UNIQUE,
    Location VARCHAR(50)
);
